* ASDF Dependencies
** Utilities
- Makes easier to install latest versions of asdf plugins
#+begin_src shell
#!/bin/bash
function asdf_latest() {
    asdf plugin add $1
    asdf install $1 latest
    asdf global $1 latest
}
#+end_src
